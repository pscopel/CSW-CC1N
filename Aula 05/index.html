  class Outliner:
        def __init__(self):
            self.outline = {}
    
        def add_topic(self, topic, parent=None):
            if parent is None:
                self.outline[topic] = []
            else:
                if parent in self.outline:
                    self.outline[parent].append(topic)
                else:
                    print(f"O tópico '{parent}' não existe!")
    
        def remove_topic(self, topic):
            if topic in self.outline:
                del self.outline[topic]
                for parent, subtopics in self.outline.items():
                    if topic in subtopics:
                        subtopics.remove(topic)
            else:
                print(f"O tópico '{topic}' não existe!")
    
        def display(self, outline=None, level=0):
            if outline is None:
                outline = self.outline
            for topic, subtopics in outline.items():
                print("  " * level + f"- {topic}")
                if subtopics:
                    self.display(dict.fromkeys(subtopics), level + 1)
    
    # Exemplo de uso
    outliner = Outliner()
    
    # Adicionando tópicos
    outliner.add_topic("Projeto")
    outliner.add_topic("Introdução", "Projeto")
    outliner.add_topic("Objetivos", "Projeto")
    outliner.add_topic("Metodologia", "Projeto")
    outliner.add_topic("Desenvolvimento", "Objetivos")
    
    # Exibindo o outliner
    outliner.display()
    
    # Removendo um tópico
    outliner.remove_topic("Objetivos")
    
    # Exibindo novamente após remoção
    print("\nApós remover 'Objetivos':")
    outliner.display()
